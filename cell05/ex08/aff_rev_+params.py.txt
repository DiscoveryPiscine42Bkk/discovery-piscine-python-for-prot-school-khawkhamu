import sys

def main():
    """
    Displays the command-line arguments in reverse order,
    or 'none' if there are fewer than two arguments.
    """
    if len(sys.argv) < 3:  # Check if there are at least two arguments (excluding script name)
        print("none")
    else:
        for arg in reversed(sys.argv[1:]):  # Iterate in reverse order, excluding script name
            print(arg)

if __name__ == "__main__":
    main()
