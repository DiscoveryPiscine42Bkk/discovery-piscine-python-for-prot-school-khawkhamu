# free_range.py
import sys

def main():
    """
    Constructs an array containing values between two given numbers and prints it.
    """
    if len(sys.argv) != 3:
        print("none")
        return

    try:
        start = int(sys.argv[1])
        end = int(sys.argv[2])
    except ValueError:
        print("none")
        return

    result_array = list(range(start, end + 1))
    print(result_array)

if __name__ == "__main__":
    main()
