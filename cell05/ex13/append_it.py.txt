import sys

def main():
    """
    Appends "ism" to each parameter passed as an argument, skipping those ending with "ism".
    """
    if len(sys.argv) == 1:
        print("none")
        return

    for arg in sys.argv[1:]:
        if not arg.endswith("ism"):
            print(arg + "ism")

if __name__ == "__main__":
    main()
