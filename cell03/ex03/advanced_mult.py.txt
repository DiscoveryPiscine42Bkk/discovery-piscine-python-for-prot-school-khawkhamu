#!/usr/bin/python3

def advanced_mult():
    """
    Displays multiplication tables from 0 to 10 using only two while loops.
    """
    table_num = 0
    while table_num <= 10:
        print(f"Table de {table_num}: ", end="")
        multiplier = 0
        while multiplier <= 10:
            print(table_num * multiplier, end=" ")
            multiplier += 1
        print()  # Move to the next line for the next table
        table_num += 1

if __name__ == "__main__":
    advanced_mult()
